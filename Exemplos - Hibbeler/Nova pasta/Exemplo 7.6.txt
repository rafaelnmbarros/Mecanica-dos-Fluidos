    "import numpy as np\n",
    "import matplotlib.pyplot as plt\n",
    "\n",
    "#Domínio de coordenadas\n",
    "xmax=5\n",
    "ymax=5\n",
    "\n",
    "\n",
    "#Plotagem do mapa de cores da função de corrente\n",
    "#====================================================\n",
    "x=np.linspace(0,xmax,100)          #Coordenadas x dos pontos da malha\n",
    "y=np.linspace(0,ymax,100)          #Coordenadas y dos pontos da malha\n",
    "X,Y = np.meshgrid(x,y)             #Criação da malha\n",
    "corrente=5*X*Y                     #Função de corrente\n",
    "fig = plt.figure(figsize=(8, 6))   #Tamanho da figura\n",
    "ax = fig.add_subplot(111)          #Apenas uma figura\n",
    "im = ax.pcolormesh(X, Y, \n",
    "    corrente, cmap='jet')          #Plotagem da função de corrente\n",
    "fig.colorbar(im)                   #Barra de cores\n",
    "\n",
    "#Plotagem de vetores velocidade\n",
    "#====================================================\n",
    "x=np.linspace(0,xmax,15)          #Coordenadas x dos pontos da malha\n",
    "y=np.linspace(0,ymax,15)          #Coordenadas y dos pontos da malha\n",
    "X,Y = np.meshgrid(x,y)            #Criação da malha\n",
    "U=5*X                             #Componente x da velocidade\n",
    "V=-5*Y                            #Componente y da velocidade\n",
    "VR=(U**2+V**2)**0.5\n",
    "ax.quiver(X, Y, U, V,  #Plotagem do vetor velocidade\n",
    "    angles='xy', scale=250, width=0.004)\n",
    "\n",
    "#Edição do gráfico\n",
    "#====================================================\n",
    "plt.title('Escoamento em torno de um canto em 90°', fontsize=20)\n",
    "plt.xlabel('x (m)', fontsize=20)\n",
    "plt.ylabel('y (m)', fontsize=20)"
